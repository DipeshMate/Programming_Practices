#include <stdio.h>
int main() 
{ 
int trainee[3][3];
int average[3] = {0};
int i, j, max=0;
for(i=0; i&lt;3; i++)
{ 
   for(j=0; j&lt;3; j++) 
{scanf("%d",&amp;trainee[i][j]);
if(trainee[i][j]&lt;1 || trainee[i][j]&gt;100)
 {trainee[i][j] = 0;} 
}
 }
for(i=0; i&lt;3; i++)

{
for(j=0; j&lt;3; j++)
 {average[i] = average[i] + trainee[j][i];}
average[i] = average[i] / 3;} 

for(i=0; i&lt;3; i++) 
{ if(average[i]&gt;max) 
{ max = average[i];}
} 
for(i=0; i&lt;3; i++) 
{ 
if(average[i]==max) 
{ 
printf("Trainee Number : %d\n",i+1);
}
if(average[i]&lt;70)
{
printf("Trainee is Unfit");
}
}
return 0;
